// MakeTestTimeEndpoint - returns endpoint to  function
func MakeTestTimeEndpoint(
	s Service,
	l log.Logger,
	mw []endpoint.Middleware,
) endpoint.Endpoint {

	var e endpoint.Endpoint
	e = makeTestTimeEndpoint(s)
	// add middlewares
	for _, m := range mw {
		e = m(e)
	}
	e = LoggingMiddleware(log.With(l, "method", ""))(e)
	return e
}

func makeTestTimeEndpoint(s Service) endpoint.Endpoint {
	return func(ctx context.Context, request interface{}) (interface{}, error) {
		req, ok := request.(models.TestTimeRequest)
		if !ok {
			return nil, ErrBadRequest
		}
		return s.TestTime(ctx, req)
	}
}
