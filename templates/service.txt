
func (s service) TestTime(
	ctx context.Context,
	req models.TestTimeRequest,
) (models.TestTimeResponse, error) {
	var (
		response models.TestTimeResponse
		err      error
	)

	if req.Str == "" {
		err = errors.New("no string was passed")
		s.logger.Log("err", "boo")
		return response, err
	}

	product := req.Str + ""
	response.Res = product
	s.logger.Log("response", response.Res)

	return response, err
}
