
// TestTime - intrumentation for foo endpoint
func (i instrumentingMiddleware) TestTime (
	ctx context.Context,
	request models.TestTimeRequest,
) (res models.TestTimeResponse, err error) {
	defer func(begin time.Time) {
		i.duration.With(
			"method", "TestTime",
			"result", fmt.Sprint(err == nil),
			"mtype", "rate",
			"unit", "s",
		).Observe(time.Since(begin).Seconds())
		i.count.With(
			"method", "TestTime",
			"result", fmt.Sprint(err == nil),
			"mtype", "count",
			"unit", "req",
		).Add(1)
	}(time.Now())
	return i.next.TestTime(ctx, request)
}
